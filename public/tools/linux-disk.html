CommandDescription
sudo fdisk -lList all disks and partitions
sudo fdisk /dev/sdaOpen fdisk for disk /dev/sda
mPrint help inside fdisk
nCreate new partition
dDelete partition
wWrite changes and exit
CommandDescription
sudo parted /dev/sda printShow partitions
sudo parted /dev/sda mklabel gptCreate GPT partition table
sudo parted /dev/sda mkpart primary ext4 1MiB 100%Create partition
CommandDescription
mount /dev/sdb1 /mntMount partition to directory
umount /mntUnmount mounted partition
mountShow all mounted filesystems
CommandDescription
sudo mkfs.ext4 /dev/sdb1Format partition as ext4 filesystem
sudo fsck /dev/sdb1Check and repair filesystem
CommandDescription
sudo lvcreate -L 10G -n myvolume myvgCreate 10GB logical volume
sudo vgextend myvg /dev/sdb1Add physical volume to volume group
sudo lvremove /dev/myvg/myvolumeRemove logical volume
